<script>

    // ----- БАЗА ДАННЫХ: русский вопрос -> правильный английский (с учётом допустимых вариантов)

    const questionsDB = [

        { ru: "Она сейчас дома?", en: "Is she at home now?" },

        { ru: "Они любят кофе?", en: "Do they like coffee?" },

        { ru: "Твой брат умеет водить машину?", en: "Can your brother drive?" },

        { ru: "Вы были вчера в офисе?", en: "Were you in the office yesterday?" },

        { ru: "Будет ли дождь завтра?", en: "Will it rain tomorrow?" },

        { ru: "Сколько времени ты ждал такси?", en: "How long did you wait for the taxi?" },

        { ru: "Я хочу спросить, почему ты передумал.", en: "I want to ask why you changed your mind." },

        { ru: "Ты не мог бы объяснить, как это работает?", en: "Could you explain how this works?" },

        { ru: "Ты не знаешь, который час?", en: "Do you know what time it is?" },

        { ru: "Ты хотел бы заказать пиццу или пасту?", en: "Would you like to order pizza or pasta?" },

        { ru: "Зачем ты купил этот старый телефон?", en: "Why did you buy that old phone?" },

        { ru: "Ты уверен, что он нас слышит?", en: "Are you sure that he can hear us?" },

        { ru: "Тебе нравится жить в этом районе?", en: "Do you like living in this area?" },

    ];


    // ----- Вспомогательные функции нормализации (сравниваем без учёта лишних пробелов, регистра, знаков)

    function normalize(str) {

        return str.toLowerCase()

                  .replace(/[^\w\s']/g, '')   // убираем пунктуацию, кроме апострофа

                  .replace(/\s+/g, ' ')

                  .trim();

    }


    function isAnswerCorrect(userInput, correctAnswer) {

        const normUser = normalize(userInput);

        const normCorrect = normalize(correctAnswer);

        return normUser === normCorrect;

    }


    // ----- Управление состоянием

    let currentQuestionsList = [];   // текущий порядок вопросов (перемешанный или нет)

    let currentIndex = 0;

    let score = 0;

    let answered = false;            // уже отвечено на текущий вопрос (блокируем повторную проверку)

    let currentCorrectAnswer = "";


    // DOM элементы

    const russianQuestionEl = document.getElementById("russianQuestion");

    const userAnswerInput = document.getElementById("userAnswer");

    const checkBtn = document.getElementById("checkBtn");

    const nextBtn = document.getElementById("nextBtn");

    const resetBtn = document.getElementById("resetBtn");

    const shuffleBtn = document.getElementById("shuffleBtn");

    const feedbackArea = document.getElementById("feedbackArea");

    const currentIndexDisplay = document.getElementById("currentIndexDisplay");

    const totalCountSpan = document.getElementById("totalCount");

    const scoreCountSpan = document.getElementById("scoreCount");


    function updateUIStats() {

        currentIndexDisplay.innerText = currentIndex + 1;

        totalCountSpan.innerText = currentQuestionsList.length;

        scoreCountSpan.innerText = score;

    }


    function loadQuestion() {

        // сбрасываем состояние ответа на текущий вопрос

        answered = false;

        const questionObj = currentQuestionsList[currentIndex];

        if (!questionObj) return;

        russianQuestionEl.innerText = questionObj.ru;

        currentCorrectAnswer = questionObj.en;

        userAnswerInput.value = "";

        // убираем стилизацию обратной связи, возвращаем нейтральное сообщение

        feedbackArea.innerHTML = `✨ Напишите перевод и нажмите «Проверить».<br><span class="tip">📌 Подсказка: ${getHintForQuestion(questionObj.ru)}</span>`;

        feedbackArea.className = "feedback";

        userAnswerInput.focus();

        updateUIStats();

    }


    // Умные подсказки на основе типа вопроса (без hardcode, по ключевым словам)

    function getHintForQuestion(russianText) {

        if (russianText.includes("Скажите, пожалуйста") || russianText.includes("не знаешь, во сколько") || russianText.includes("не мог бы объяснить") || russianText.includes("я хочу спросить") || russianText.includes("который час"))

            return "Непрямой вопрос → порядок слов как в утверждении: подлежащее + сказуемое (без do/does/did).";

        if (russianText.match(/^(Ты|Она|Они|Вы|Это|У тебя|Твой брат)/) && !russianText.match(/где|когда|почему|как|сколько|какой|зачем|во сколько/))

            return "Общий вопрос → начни с глагола: Do/Does/Is/Are/Can/Have/Were/Will.";

        if (russianText.match(/где|когда|почему|как часто|кого|какой|сколько|зачем|как долго|во сколько/))

            return "Специальный вопрос: вопросительное слово + вспомогательный глагол + подлежащее + смысловой глагол.";

        return "Проверь порядок слов: вспомогательный глагол перед подлежащим (кроме непрямых).";

    }


    function checkAnswer() {

        if (answered) {

            feedbackArea.innerHTML = "⚠️ Вы уже ответили на этот вопрос. Нажмите «Следующий» или перемешайте.";

            feedbackArea.className = "feedback wrong";

            return;

        }

        const userAnswer = userAnswerInput.value.trim();

        if (userAnswer === "") {

            feedbackArea.innerHTML = "❓ Введите перевод на английский.";

            feedbackArea.className = "feedback wrong";

            return;

        }

        const correct = isAnswerCorrect(userAnswer, currentCorrectAnswer);

        if (correct) {

            // засчитываем только если ещё не отвечено

            score++;

            scoreCountSpan.innerText = score;

            feedbackArea.innerHTML = `✅ Отлично! «${userAnswer}» — верно!<br><span class="tip">🎉 Правильный порядок слов.</span>`;

            feedbackArea.className = "feedback correct";

            answered = true;

        } else {

            feedbackArea.innerHTML = `❌ Неверно.<br><strong>Ваш ответ:</strong> «${userAnswer}»<br><strong>✅ Правильно:</strong> «${currentCorrectAnswer}»<br><span class="tip">🔁 Совет: ${getHintForQuestion(currentQuestionsList[currentIndex].ru)}</span>`;

            feedbackArea.className = "feedback wrong";

            answered = true;   // блокируем возможность повторно набрать очки, но показываем верный ответ

        }

        updateUIStats();

    }


    function nextQuestion() {

        if (currentIndex + 1 < currentQuestionsList.length) {

            currentIndex++;

            loadQuestion();

        } else {

            // если это был последний вопрос

            if (currentIndex === currentQuestionsList.length - 1) {

                feedbackArea.innerHTML = `🏆 Поздравляю! Вы прошли все ${currentQuestionsList.length} вопросов. Нажмите «Перемешать», чтобы начать заново, или сбросьте счет. 🏆`;

                feedbackArea.className = "feedback correct";

            } else {

                loadQuestion();

            }

        }

    }


    function resetScore() {

        score = 0;

        scoreCountSpan.innerText = score;

        answered = false;

        // но текущий вопрос не меняем, просто обнуляем очки

        feedbackArea.innerHTML = `🧹 Счёт сброшен. Продолжайте отвечать на текущий вопрос.`;

        feedbackArea.className = "feedback";

        if (!answered) userAnswerInput.value = "";

        updateUIStats();

    }


    function shuffleQuestions() {

        // перемешиваем массив (Fisher-Yates)

        const newList = [...currentQuestionsList];

        for (let i = newList.length - 1; i > 0; i--) {

            const j = Math.floor(Math.random() * (i + 1));

            [newList[i], newList[j]] = [newList[j], newList[i]];

        }

        currentQuestionsList = newList;

        currentIndex = 0;

        answered = false;

        if (score !== undefined) score = score;   // сохраняем счёт при перемешивании, но можно по желанию оставить.

        // однако студенты могут накручивать, но так логичнее — счёт продолжает копиться.

        loadQuestion();

        feedbackArea.innerHTML = "🔀 Вопросы перемешаны. Продолжайте тренировку! Счёт сохранён.";

        feedbackArea.className = "feedback";

    }


    function resetAndStartFresh() {

        // жёсткий сброс всех данных к исходному порядку (без перемешивания)

        currentQuestionsList = [...questionsDB];

        currentIndex = 0;

        score = 0;

        answered = false;

        loadQuestion();

        feedbackArea.innerHTML = "🔄 Прогресс полностью сброшен. Начинаем сначала.";

        feedbackArea.className = "feedback";

    }


    // инициализация

    function init() {

        currentQuestionsList = [...questionsDB];

        currentIndex = 0;

        score = 0;

        answered = false;

        loadQuestion();

        // кнопки

        checkBtn.addEventListener("click", checkAnswer);

        nextBtn.addEventListener("click", nextQuestion);

        resetBtn.addEventListener("click", resetScore);

        shuffleBtn.addEventListener("click", shuffleQuestions);

        // обработка Enter в поле ввода

        userAnswerInput.addEventListener("keypress", function(e) {

            if (e.key === "Enter") {

                e.preventDefault();

                if (!answered) {

                    checkAnswer();

                } else {

                    nextQuestion();

                }

            }

        });

    }


    init();

</script>
